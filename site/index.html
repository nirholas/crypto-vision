from xtools import XTools
from xtools.actions.types import AutoLikeConfig, LikeFilters

# Configure smart auto-liker
config = AutoLikeConfig(
    keywords=["Python", "AI", "startup"],
    max_likes_per_hour=30,          # Stay safe
    min_delay=45.0,                  # Human-like timing
    max_delay=180.0,
    skip_retweets=True,              # Only original content
    skip_verified=False,             # Include verified accounts
    engagement_threshold=50,         # Min engagement
)

# Add smart filters
filters = LikeFilters(
    min_likes=10,                    # Quality threshold
    min_followers=500,               # Avoid spam accounts  
    max_followers=100000,            # Skip mega accounts
    exclude_keywords=["crypto scam", "giveaway"],
    language="en",
)

async with XTools() as x:
    liker = await x.engage.auto_liker(config, filters)
    
    # Run with real-time analytics
    async for result in liker.run():
        print(f"✓ Liked: @{result.author} - {result.engagement_rate}%")
        
        # Stop if rate limited
        if result.rate_limited:
            print("⚠️ Rate limited, cooling down...")
            await asyncio.sleep(900)  # 15 min cooldown

from xtools import XTools
from xtools.ai import SmartTargeting
from xtools.analytics import GrowthTracker

# AI-powered growth automation
targeting = SmartTargeting(
    niche="tech startups",
    ideal_follower_profile={
        "interests": ["SaaS", "AI", "Python"],
        "follower_range": (1000, 50000),
        "engagement_rate_min": 2.0,
        "account_age_months": 6,
    }
)

async with XTools() as x:
    tracker = GrowthTracker()
    
    while True:
        # Find ideal accounts to engage with
        targets = await targeting.find_targets(
            seed_accounts=["@elonmusk", "@sama", "@paulg"],
            strategy="engaged_followers",  # People who reply
            limit=20
        )
        
        for target in targets:
            # Multi-touch engagement strategy
            await x.engage.like_recent(target.username, count=3)
            await x.follow.user(target.username)
            
            # Optionally reply to their pinned tweet
            if target.pinned_tweet and target.engagement_score > 80:
                reply = await ai.generate_reply(target.pinned_tweet)
                await x.engage.reply(target.pinned_tweet.url, reply)
            
            tracker.log_engagement(target)
        
        # Analytics dashboard update
        stats = tracker.get_daily_stats()
        print(f"📊 Today: +{stats.new_followers} followers, "
              f"{stats.engagement_rate}% engagement")
        
        await asyncio.sleep(3600)  # Run hourly

🤖 DETECTION

Bot Army Detector

Identify fake followers and bot networks. Clean your follower list and avoid fake engagement.

from xtools.ai import BotDetector

detector = BotDetector()
followers = await x.scrape.followers("me")

bots = []
for follower in followers:
    score = await detector.analyze(follower)
    if score.bot_probability > 0.8:
        bots.append(follower)
        
print(f"Found {len(bots)} likely bots")

Multi-Account Orchestra

Manage multiple accounts with different personas. Each account can have its own engagement style, target audience, and automation rules. Perfect for agencies or brand management.

from xtools import XTools, AccountManager

manager = AccountManager()

# Load multiple accounts
accounts = [
    manager.load_account("tech_account", persona="developer"),
    manager.load_account("news_account", persona="journalist"),
    manager.load_account("personal", persona="casual"),
]

# Each runs with its own style
async with XTools(accounts) as x:
    for account in x.accounts:
        # AI adapts tone based on persona
        await account.engage.auto_like(
            keywords=account.persona.interests,
            style=account.persona.engagement_style
        )
